// ID validation
=AND(NOT(ISNUMBER(VALUE(LEFT(B2,1)))),NOT(ISNUMBER(VALUE(MID(B2,2,1)))),IF(LEN(B2) > 3, True, False),MID(B2,3,1) = "_",EXACT(B2,UPPER(B2)))

// Email function
=AND(ISERROR(FIND(" ",L2)),LEN(L2)-LEN(SUBSTITUTE(L2,"@",""))=1,IFERROR(SEARCH("@",L2)<SEARCH(".",L2,SEARCH("@",L2)),0),NOT(IFERROR(SEARCH("@",L2),0)=1),NOT(IFERROR(SEARCH(".",L2,SEARCH("@",L2))-SEARCH("@",L2),0)=1),LEFT(L2,1)<>".",RIGHT(L2,1)<>".")
=AND(ISERROR(FIND(" ",M2)),LEN(M2)-LEN(SUBSTITUTE(M2,"@",""))=1,IFERROR(SEARCH("@",M2)<SEARCH(".",M2,SEARCH("@",M2)),0),NOT(IFERROR(SEARCH("@",M2),0)=1),NOT(IFERROR(SEARCH(".",M2,SEARCH("@",M2))-SEARCH("@",M2),0)=1),LEFT(M2,1)<>".",RIGHT(M2,1)<>".")

// Phone validation
=OR(AND(LEFT(K2,1) = "+", ISNUMBER(VALUE(MID(K2, 2,LEN(K2))))), AND(LEFT(K2,2) = "00",ISNUMBER(VALUE(K2))))
